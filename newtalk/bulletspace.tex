\usepackage{xpatch}
\xpatchcmd{\itemize}
{\def\makelabel}
	{\ifnum\@itemdepth=1\relax
		\setlength\itemsep{3ex plus 4ex minus 3ex}% separation for first level
	\else
	\ifnum\@itemdepth=2\relax
		\setlength\itemsep{1.5ex plus 2ex minus 1.5ex}% separation for second level
	\else
	\ifnum\@itemdepth=3\relax
		\setlength\itemsep{1ex plus 1ex minus 1ex}% separation for third level
\fi\fi\fi\def\makelabel
}{}{}
